const SHEET_NAME = 'Records'; function doGet(e) { const action = e.parameter.action; if (action === 'getPersons') return jsonResponse(getPersons()); if (action === 'getRecords') return jsonResponse(getRecords(e.parameter.personId)); return jsonResponse({error: 'unknown action'}); } function doPost(e) { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'saveRecord') return jsonResponse(saveRecord(data.record)); if (action === 'savePerson') return jsonResponse(savePerson(data.person)); if (action === 'deletePerson') return jsonResponse(deletePerson(data.personId)); if (action === 'updateRecord') return jsonResponse(updateRecord(data.record)); function updateRecord(rec) { const sheet = getSheet(SHEET_NAME); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === rec.id) { sheet.getRange(i+1,1,1,17).setValues([[ rec.id, rec.personId, rec.personName, rec.date, rec.height, rec.weight, rec.temp, rec.meals?.breakfast||'', rec.meals?.lunch||'', rec.meals?.dinner||'', rec.meals?.snack||'', rec.proc||'', rec.fried||'', rec.med||'', (rec.statuses||[]).join(','), rec.otherText||'', rec.mens||'' ]]); return {ok: true}; } } return {error: 'not found'}; } return jsonResponse({error: 'unknown action'}); } function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); return ss.getSheetByName(name) || ss.insertSheet(name); } function getPersons() { const sheet = getSheet('Persons'); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; return data.slice(1).map(r => ({id:r[0], name:r[1], color:r[2]})); } function savePerson(person) { const sheet = getSheet('Persons'); if (sheet.getLastRow() === 0) sheet.appendRow(['id','name','color']); sheet.appendRow([person.id, person.name, person.color]); return {ok: true}; } function deletePerson(personId) { const sheet = getSheet('Persons'); const data = sheet.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (data[i][0] === personId) { sheet.deleteRow(i + 1); break; } } const rSheet = getSheet(SHEET_NAME); const rData = rSheet.getDataRange().getValues(); for (let i = rData.length - 1; i >= 1; i--) { if (rData[i][1] === personId) rSheet.deleteRow(i + 1); } return {ok: true}; } function getRecords(personId) { const sheet = getSheet(SHEET_NAME); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; return data.slice(1) .filter(r => !personId || r[1] === personId) .map(r => ({ id:r[0], personId:r[1], personName:r[2], date:r[3], height:r[4], weight:r[5], temp:r[6], meals:{breakfast:r[7],lunch:r[8],dinner:r[9],snack:r[10]}, proc:r[11], fried:r[12], med:r[13], statuses:r[14]?r[14].split(','):[], otherText:r[15], mens:r[16]||null })); } function saveRecord(rec) { const sheet = getSheet(SHEET_NAME); if (sheet.getLastRow() === 0) { sheet.appendRow(['id','personId','personName','date', 'height','weight','temp','breakfast','lunch','dinner','snack', 'proc','fried','med','statuses','otherText','mens']); } sheet.appendRow([ rec.id, rec.personId, rec.personName, rec.date, rec.height, rec.weight, rec.temp, rec.meals?.breakfast||'', rec.meals?.lunch||'', rec.meals?.dinner||'', rec.meals?.snack||'', rec.proc||'', rec.fried||'', rec.med||'', (rec.statuses||[]).join(','), rec.otherText||'', rec.mens||'' ]); return {ok: true}; } function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
